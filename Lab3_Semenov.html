knn = KNeighborsClassifier(n_neighbors=5)

roc_auc_knn = roc_auc_score(y_test, y_pred_proba_knn)

print(f"Accuracy:  {accuracy_knn:.4f}")
print(f"Precision: {precision_knn:.4f}")
print(f"Recall:    {recall_knn:.4f}")
print(f"F1-score:  {f1_knn:.4f}")
print(f"ROC-AUC:   {roc_auc_knn:.4f}")

print("\n  :")
print(classification_report(y_test, y_pred_knn, target_names=['', '']))

#   KNN
fig, axes = plt.subplots(1, 3, figsize=(15, 4))

#  
cm_knn = confusion_matrix(y_test, y_pred_knn)
disp_knn = ConfusionMatrixDisplay(confusion_matrix=cm_knn, display_labels=['', ''])
disp_knn.plot(ax=axes[0], cmap='Blues')
axes[0].set_title('  (KNN)')

# ROC-
fpr_knn, tpr_knn, _ = roc_curve(y_test, y_pred_proba_knn)

axes[1].set_xlabel('False Positive Rate')
axes[1].set_ylabel('True Positive Rate')
axes[1].set_title('ROC- (KNN)')
axes[1].legend()
axes[1].grid(alpha=0.3)

#   k  accuracy
k_values = range(1, 21)
knn_accuracies = []

for k in k_values:
    knn_temp = KNeighborsClassifier(n_neighbors=k)
    knn_temp.fit(X_train, y_train)
    y_pred_temp = knn_temp.predict(X_test)
    knn_accuracies.append(accuracy_score(y_test, y_pred_temp))

               label=f' k={best_k}')
